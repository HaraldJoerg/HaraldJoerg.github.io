light_source(0,5,10);
background(0.8,0.8,1.0);
camera(position => [0.5,0.5,3], look_at => [0.5,0.5,0.5]);
my $n = 5;
for my $x (0..$n) {
    for my $y (0..$n) {
        for my $z (0..$n) {
            sphere(1/(3*$n))
                ->rgb($x/$n,$y/$n,$z/$n)
                ->translate($x/$n,$y/$n,$z/$n);
        }
    }
}
